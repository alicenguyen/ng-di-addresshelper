    OnInit,
    Input,

    AfterViewInit,

    ChangeDetectorRef

} from '@angular/core';
import {Address} from '../address.model';
import {FormControl, FormGroup, FormBuilder, Validators} from "@angular/forms";
import {debug} from "util";
import {Observable} from "rxjs/Observable";
import {AddressHelperService} from "../address-helper.service";


@Component({
    selector: 'zen-address-form',
    templateUrl: './address-form.component.html',
    styleUrls: ['./address-form.component.css']
})

export class AddressFormComponent implements OnChanges, OnInit, AfterViewChecked, AfterViewInit {
    // heey, i'm letting my mom say waht this variable is
    @Input() inputFormValue: any = "meowwww<3";

    model: {

        zip: any

    address2: string = "2A";

    city: string = "Drammen";

    zip: number = 3047;

    nextId: number = 0;


    @Input()
    selectionCache: any;

    // this.SelectionCache
    // this.testForm.get('selectionCache).value;
    _cache: any;
    testForm: FormGroup;

    diAddressHelper: any;

    constructor(private _fb: FormBuilder, private ref: ChangeDetectorRef, private _diah: AddressHelperService) {
        this.testForm = this._fb.group({
            origin: ['', Validators.required],
            destination: ['', Validators.required],
            orderDate: ['', Validators.required]
        });

        this.diAddressHelper = _diah
    }


    ngOnInit() {
        this.logIt(`OnInit`)
        this.diAddressHelper.init()
        this.diAddressHelper.listen(function() {
            console.log('booya!')
        })

    }

    ngAfterViewInit() {
        this.logIt(`ngAfterViewInit()`)
    }

    ngAfterViewChecked(): void {

    }

    ngOnChanges(changes: SimpleChanges): void {
        this.logIt(`ngOnChanges`);

        const cache: SimpleChange = changes.selectionCache
        // this.model = new Address(this.address, this.city, this.zip)
        this.logIt(`prev value: ${cache.previousValue}`);
        this.logIt(`curr value: ${cache.currentValue}`);
        this._cache = cache.currentValue
        //
        // let address = cache.currentValue
        // debugger
        // this.form['address1'] = address.street.streetName
        // this.form['address2'] = address.streetNumber.streetNo
        // this.form['city'] = address.street.city
        // this.form['zip'] = address.streetNumber.postalCode
    }

    logIt(msg: string) {
        console.log(`#${this.nextId++} ${msg}`);
    }

    onChange(event: Event) {
        event.preventDefault()
        this.logIt('onChange')

    }


}
